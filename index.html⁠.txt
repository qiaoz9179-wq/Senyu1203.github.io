        // Price Logic engine
        function calculatePrice() {
            let total = 0;
            let listItems = [];
            let tips = [];

            // 1. Check Gel Service & Base Price
            const mainCat = document.getElementById('main-category').value;
            let gelPrice = 0;
            let gelName = '';
            const hasGel = mainCat !== 'none';

            if (hasGel) {
                if (mainCat === 'special-monthly') {
                    gelPrice = 1299;
                    gelName = '當月特價款';
                } else {
                    const selectElem = document.getElementById('sub-choice');
                    const selectedText = selectElem.options[selectElem.selectedIndex].text;
                    gelPrice = parseInt(selectElem.value);
                    gelName = selectedText.split('—')[0].trim();
                }
            }

            // 2. High Price Jump Color Logic (以高價款計價)
            const jumpColorQty = counters['jump-color'];
            let jumpColorExtraCost = 0;

            if (jumpColorQty > 0 && hasGel) {
                const highTierSelect = document.getElementById('jump-color-high-tier');
                const highTierPrice = parseInt(highTierSelect.value) || 0;

                if (highTierPrice > gelPrice) {
                    const priceDiff = highTierPrice - gelPrice;
                    gelPrice = highTierPrice; 
                    gelName += ` (配合高價跳色升級)`;
                    tips.push(`💡 **跳色高價計費**：整手底色調高至 $${highTierPrice} 計算（高低差 +$${priceDiff}）。`);
                }
                jumpColorExtraCost = jumpColorQty * 50;
            }

            // Push base gel service if active
            if (hasGel) {
                let finalBasePrice = gelPrice;
                if (selectedTarget === 'foot') {
                    finalBasePrice += 100;
                    gelName += ' (含足部 +$100)';
                }
                listItems.push({ name: `💅 ${gelName}`, price: finalBasePrice });
                total += finalBasePrice;

                    tips.push("✨ 純卸甲加購足部基礎保養，現省 $150（半價優惠）！");
                } else {
                    listItems.push({ name: `🦶 足部基礎保養`, price: 300 });
                    total += 300;
                }
            }

            // Cache data before discount
            lastGeneratedReceiptItems = [...listItems];
            lastGeneratedTotalBeforeDiscount = total;

            // 6. Manual Custom Discount Calculation (自訂優惠折抵)
            const discountVal = parseInt(document.getElementById('cust-discount').value) || 0;
            let finalTotal = Math.max(0, total - discountVal);
            lastGeneratedFinalTotal = finalTotal;

            if (discountVal > 0) {
                // Display discount in real-time items list too
                listItems.push({ name: `🎁 專屬特惠折抵金額`, price: -discountVal });
            }

            // Render items in UI
            renderReceipt(listItems);
            
            // Show / Hide Tips
            const tipsContainer = document.getElementById('calc-rules-tips');
            const tipsContent = document.getElementById('rules-tips-content');
            if (tips.length > 0) {
                tipsContainer.classList.remove('hidden');
                tipsContent.innerHTML = tips.map(t => `<p class="leading-relaxed">${t}</p>`).join('');
            } else {
                tipsContainer.classList.add('hidden');
            }

            // Set Final Total
            document.getElementById('grand-total').textContent = `$${finalTotal}`;

            // Generate Clipboard Text Previews
            updateClipboardText(listItems, finalTotal, discountVal);
        }

        // Render receipt logic helper
        function renderReceipt(items) {

            if (items.length === 0) {

        // Generate and sync the final Line formatted quote string
        function updateClipboardText(items, total, discountVal) {
            const txtArea = document.getElementById('quote-clipboard-textarea');
            const custName = document.getElementById('cust-name').value.trim();
            const custPhone = document.getElementById('cust-phone').value.trim();
            const custDateInput = document.getElementById('cust-date').value;
            const colorCode = document.getElementById('cust-color-code').value.trim();

            let formattedDate = "";
            if (custDateInput) {
                const d = new Date(custDateInput);
                formattedDate = `${d.getFullYear()}/${(d.getMonth()+1).toString().padStart(2,'0')}/${d.getDate().toString().padStart(2,'0')} ${d.getHours().toString().padStart(2,'0')}:${d.getMinutes().toString().padStart(2,'0')}`;
            }

            if (items.length === 0 && !custName && !custPhone && !colorCode) {
                txtArea.value = "尚未選取服務或輸入顧客資料。";
                return;
            }

            let text = `🌿 森語美甲 Forest Whisper 🌿\n`;
            text += `【預約顧客紀錄與報價】\n`;
            text += `---------------------------\n`;
            if (custName)  text += `👤 顧客姓名：${custName}\n`;
            if (custPhone) text += `📞 聯絡電話：${custPhone}\n`;
            if (formattedDate) text += `📅 預約時間：${formattedDate}\n`;
            if (colorCode) text += `🎨 施作色號：${colorCode}\n`;
            text += `💳 付款方式：${paymentMethod}\n`;
            text += `---------------------------\n`;
            
            // List services
            let itemIndex = 1;
            lastGeneratedReceiptItems.forEach(item => {
                text += `${itemIndex++}. ${item.name} ：$${item.price}\n`;
            });

            if (discountVal > 0) {
                text += `🎁 優惠折扣折抵：-$${discountVal}\n`;
            }

            text += `---------------------------\n`;
            text += `💰 實收總金額：$${total} 元\n`;
            text += `\n*此報價依 1/13 新版官方價目表精確計算。`;
            text += `\n期待在溫慢的森林密語中與您相遇！🍃`;

            txtArea.value = text;
        }

        // ==================== CLOUD DATABASE STORAGE SYSTEM ====================
        function saveAndCopyQuote() {
            if (lastGeneratedReceiptItems.length === 0) {
                alert('請先選取服務項目再進行儲存！');
                return;
            }

            const custName = document.getElementById('cust-name').value.trim() || "現場散客";
            const custPhone = document.getElementById('cust-phone').value.trim() || "未留電話";
            const custDateInput = document.getElementById('cust-date').value;
            const colorCode = document.getElementById('cust-color-code').value.trim() || "無備註";
            const discountVal = parseInt(document.getElementById('cust-discount').value) || 0;

            let formattedDate = "";
            if (custDateInput) {
                const d = new Date(custDateInput);
                formattedDate = `${d.getFullYear()}/${(d.getMonth()+1).toString().padStart(2,'0')}/${d.getDate().toString().padStart(2,'0')} ${d.getHours().toString().padStart(2,'0')}:${d.getMinutes().toString().padStart(2,'0')}`;
            } else {
                const d = new Date();
                formattedDate = `${d.getFullYear()}/${(d.getMonth()+1).toString().padStart(2,'0')}/${d.getDate().toString().padStart(2,'0')} ${d.getHours().toString().padStart(2,'0')}:${d.getMinutes().toString().padStart(2,'0')}`;
            }

            // Create record conforming to requirements (No complex queries rule observed)
            const newQuoteRecord = {
                id: 'quote-' + Date.now() + '-' + Math.floor(Math.random()*1000),
                customerName: custName,
                customerPhone: custPhone,
                appointmentDate: formattedDate,
                colorNotes: colorCode,
                payMethod: paymentMethod,
                target: selectedTarget === 'hand' ? "手部美甲" : "足部美甲",
                items: lastGeneratedReceiptItems,
                discount: discountVal,
                totalPrice: lastGeneratedFinalTotal,
                lineText: document.getElementById('quote-clipboard-textarea').value
            };

            // Save locally
            savedHistoryQuotes.unshift(newQuoteRecord);
            
            try {
                localStorage.setItem('forest_whisper_cloud_quotes_v2', JSON.stringify(savedHistoryQuotes));
            } catch(e) {
                console.error("無法備份資料至本機空間");
            }

            // Copy Quote to Clipboard
            const txtArea = document.getElementById('quote-clipboard-textarea');
            txtArea.select();
            txtArea.setSelectionRange(0, 99999);
            document.execCommand('copy');

            // Button Feedback
            const copyBtn = document.getElementById('copy-btn');
            const copyBtnSpan = copyBtn.querySelector('span');
            copyBtn.classList.replace('bg-sky-700', 'bg-emerald-600');
            copyBtnSpan.innerText = '儲存並複製成功！';

            // Show Toast Banner
            const toast = document.getElementById('copy-toast');
            toast.classList.remove('hidden');

            updateHistoryBadge();

            setTimeout(() => {
                copyBtn.classList.replace('bg-emerald-600', 'bg-sky-700');
                copyBtnSpan.innerText = '儲存並複製';
                toast.classList.add('hidden');
            }, 3000);
        }

        // Search Engine - Searches name, phone, color notes, part, items, price, payMethod

        // Direct Clipboard action from history card
        function copyQuoteTextDirectly(id) {
            const record = savedHistoryQuotes.find(q => q.id === id);
            if (!record) return;

            const tempInput = document.createElement('textarea');
            tempInput.value = record.lineText;
            document.body.appendChild(tempInput);
            tempInput.select();
            document.execCommand('copy');
            document.body.removeChild(tempInput);

            alert(`✅ 顧客「${record.customerName}」的歷史施作明細複製成功！`);
        }

        // Delete Quote record from list

            if (!confirm('您確定要刪除這筆雲端歷史紀錄嗎？（刪除後將無法復原）')) {

            updateHistoryBadge();
